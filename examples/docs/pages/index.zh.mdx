# 阿基拉

《阿基拉》（日语：AKIRA）是一部日本漫画家及动画导演大友克洋于 1988 年推出的动画电影，
改编自大友本人的原作漫画，推出后除了在西方世界引起广泛回响，并造成一波日本动画风潮。

import { useSSG } from 'nextra/ssg'

export const getStaticProps = ({ params }) => {
  return {
    props: {
      // We return an `ssg` object in the props.
      ssg: {
        generated_at: new Date().toISOString() + ' (ZH)'
      }
    },
    revalidate: 5
  }
}

export const DateRenderer = () => {
  // And later we can get it inside MDX.
  const { generated_at } = useSSG()
  return <strong>{generated_at}</strong>
}
