# Tabs

A built-in component provided by `nextra-theme-docs`.

## Example

<Nextra.Tabs items={['JavaScript', 'C++', {label:'C', disabled: true}, 'Python']} defaultIndex={1}>
  <Nextra.Tab>
    ```js filename="hi.js"
    import { useState, useEffect } from 'react';
    ```
  </Nextra.Tab>
  <Nextra.Tab>
    ```cpp filename="hi.cpp"
    #include <iostream>
    ```
  </Nextra.Tab>
  <Nextra.Tab>
    ```c filename="hi.c"
    #include <stdio.h>
    ```
  </Nextra.Tab>
  <Nextra.Tab>
     ```python filename="hello.py"
    print('Hello, world!')
    ```
  </Nextra.Tab>
</Nextra.Tabs>

## Usage

### MDX component

````mdx filename="tabs.mdx"
<Nextra.Tabs items={['JavaScript', 'C++', {label:'C', disabled: true}, 'Python']} defaultIndex={1}>
  <Nextra.Tab>
    ```js filename="hi.js"
    import { useState, useEffect } from 'react';
    ```
  </Nextra.Tab>
  <Nextra.Tab>
    ```cpp filename="hi.cpp"
    #include <iostream>
    ```
  </Nextra.Tab>
  <Nextra.Tab>
    ```c filename="hi.c"
    #include <stdio.h>
    ```
  </Nextra.Tab>
  <Nextra.Tab>
    ```python filename="hello.py"
    print('Hello, world!')
    ```
  </Nextra.Tab>
</Nextra.Tabs>
````

### React Component

```jsx filename="tabs.jsx"
import { Tabs, Tab } from 'nextra-docs-theme/Tabs'
const items = ['1', '2', '3', '4']
const defaultIndex = 1
const Component = () => (
  <Tabs items={items} defaultIndex={defaultIndex}>
    <Tabs.Tab>1</Tab>
    <Tabs.Tab>2</Tab>
    <Tabs.Tab>3</Tab>
    <Tabs.Tab>4</Tab>
  </Tabs>
)
export default Component
```
